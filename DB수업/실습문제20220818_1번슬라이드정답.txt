--순차적으로 증가하는 숫자
--drop table song;
create table song
( i_d number primary key, title varchar2(100), price number);
--drop table singer;
create table singer 
(id number primary key, name varchar2(100), hit_song_id number,
foreign key(hit_song_id) references song(i_d));
